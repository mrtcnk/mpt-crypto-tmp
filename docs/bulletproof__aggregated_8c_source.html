382int fold_generators(
383 const secp256k1_context* ctx,
384 secp256k1_pubkey* final_point,
385 const secp256k1_pubkey* generators,
386 const unsigned char* u_flat,
387 const unsigned char* uinv_flat,
388 size_t n,
389 size_t rounds,
390 int is_H /* 0 = G folding, 1 = H folding */
391) {
392 /* n must be power-of-two and rounds must match log2(n) */
393 if (n == 0 || (n & (n - 1)) != 0) return 0;
394 if (((size_t)1 << rounds) != n) return 0;
395
396 /* Allocate scalars for MSM: n * 32 bytes */
397 unsigned char* s_flat = (unsigned char*)malloc(n * 32);
398 if (!s_flat) return 0;
399
400 unsigned char current_s[32];
401 int ok = 0;
402
403 for (size_t i = 0; i < n; i++) {
404 /* current_s = 1 */
405 memset(current_s, 0, 32);
406 current_s[31] = 1;
407
408 for (size_t j = 0; j < rounds; j++) {
409 /* bit from MSB to LSB across 'rounds' bits */
410 int bit = (int)((i >> (rounds - 1 - j)) & 1);
411
412 const unsigned char* uj = u_flat + 32 * j;
413 const unsigned char* ujinv = uinv_flat + 32 * j;
414
415 if (!is_H) {
416 /* G folding: bit 0 -> u_inv, bit 1 -> u */
417 secp256k1_mpt_scalar_mul(current_s, current_s, bit ? uj : ujinv);
418 } else {
419 /* H folding: bit 0 -> u, bit 1 -> u_inv */
420 secp256k1_mpt_scalar_mul(current_s, current_s, bit ? ujinv : uj);
421 }
422 }
423
424 memcpy(s_flat + (i * 32), current_s, 32);
425 }
426
427 ok = secp256k1_bulletproof_ipa_msm(ctx, final_point, generators, s_flat, n);
428
429 OPENSSL_cleanse(current_s, 32);
430 OPENSSL_cleanse(s_flat, n * 32);
431 free(s_flat);
432
433 return ok;
434}
435/*
436 * Apply verifier-side IPA updates to P for `rounds` rounds.
437 * Update rule per round i:
438 * P <- P + (u_i^2) * L_i + (u_i^{-2}) * R_i
439 * u_flat / uinv_flat are (rounds * 32)-byte arrays:
440 * u_i = u_flat + 32*i
441 * u_iinv = uinv_flat + 32*i
442 */
443int apply_ipa_folding_to_P(
444 const secp256k1_context* ctx,
445 secp256k1_pubkey* P,
446 const secp256k1_pubkey* L_vec,
447 const secp256k1_pubkey* R_vec,
448 const unsigned char* u_flat,
449 const unsigned char* uinv_flat,
450 size_t rounds
451) {
452 unsigned char u_sq[32], uinv_sq[32];
453 secp256k1_pubkey tL, tR;
454 const secp256k1_pubkey* pts[3];
455
456 for (size_t i = 0; i < rounds; i++) {
457 const unsigned char* ui = u_flat + 32 * i;
458 const unsigned char* uiinv = uinv_flat + 32 * i;
459
460 /* u_sq = u_i^2, uinv_sq = (u_i^{-1})^2 = u_i^{-2} */
461 secp256k1_mpt_scalar_mul(u_sq, ui, ui);
462 secp256k1_mpt_scalar_mul(uinv_sq, uiinv, uiinv);
463
464 /* tL = (u_i^2) * L_i */
465 tL = L_vec[i];
466 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tL, u_sq)) {
467 OPENSSL_cleanse(u_sq, 32);
468 OPENSSL_cleanse(uinv_sq, 32);
469 return 0;
470 }
471
472 /* tR = (u_i^{-2}) * R_i */
473 tR = R_vec[i];
474 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tR, uinv_sq)) {
475 OPENSSL_cleanse(u_sq, 32);
476 OPENSSL_cleanse(uinv_sq, 32);
477 return 0;
478 }
479
480 /* P <- P + tL + tR */
481 pts[0] = P;
482 pts[1] = &tL;
483 pts[2] = &tR;
484
485 secp256k1_pubkey newP;
486 if (!secp256k1_ec_pubkey_combine(ctx, &newP, pts, 3)) {
487 OPENSSL_cleanse(u_sq, 32);
488 OPENSSL_cleanse(uinv_sq, 32);
489 return 0;
490 }
491 *P = newP;
492
493 }
494
495 OPENSSL_cleanse(u_sq, 32);
496 OPENSSL_cleanse(uinv_sq, 32);
497 return 1;
498}

513 const secp256k1_context* ctx,
514 secp256k1_pubkey* L,
515 secp256k1_pubkey* R,
516 const unsigned char* a_L,
517 const unsigned char* a_R,
518 const unsigned char* b_L,
519 const unsigned char* b_R,

521 const secp256k1_pubkey* G_R,
522 const secp256k1_pubkey* H_L,
523 const secp256k1_pubkey* H_R,
524 const secp256k1_pubkey* U,
525 const unsigned char* ux,

529 unsigned char cLux[32], cRux[32];

534

536 if (!secp256k1_bulletproof_ipa_dot(ctx, cL, a_L, b_R, half_n)) return 0;

570 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &term, cRux)) goto cleanup;
571 if (!add_term(ctx, &acc, &acc_inited, &term)) goto cleanup;
572 }
573
574 if (acc_inited == 0) {
575 goto cleanup;
576 }
577 *R = acc;
578
579 OPENSSL_cleanse(cL, 32);
580 OPENSSL_cleanse(cR, 32);
581 OPENSSL_cleanse(cLux, 32);
582 OPENSSL_cleanse(cRux, 32);

584

586 OPENSSL_cleanse(cL, 32);
587 OPENSSL_cleanse(cR, 32);
588 OPENSSL_cleanse(cLux, 32);
589 OPENSSL_cleanse(cRux, 32);
590 return 0;
591}

663 if (!secp256k1_ec_pubkey_combine(ctx, &H[i], pts, 2)) goto cleanup;
664 }
665
666 ok = 1;
667

669 OPENSSL_cleanse(t1, 32);
670 OPENSSL_cleanse(t2, 32);

680 *

682 * ux = H(commit_inp || <a,b>)
683 *
684 * and reused consistently throughout the IPA:
685 * - L/R cross-term construction

714 */

716 return 0;
717 }

785 if (n == 0 || (n & (n - 1)) != 0) return 0;
786
787 /* rounds = log2(n) */
788 while (cur_n > 1) { cur_n >>= 1; rounds++; }
789 cur_n = n;
790
791 /* Bounds check (CRITICAL for aggregated proofs) */
792 if (rounds > max_rounds) return 0;
793
794 /* Seed transcript */
795 memcpy(last_challenge, ipa_transcript_id, 32);
796
797 for (size_t r = 0; r < rounds; ++r) {
798 size_t half_n = cur_n >> 1;
799 secp256k1_pubkey Lr, Rr;
800
801 /* 1) Compute cross-term commitments Lr, Rr */
802 if (!secp256k1_bulletproof_ipa_compute_LR(
803 ctx, &Lr, &Rr,
804 a_vec, a_vec + half_n * 32,
805 b_vec, b_vec + half_n * 32,
806 G_vec, G_vec + half_n,
807 H_vec, H_vec + half_n,
808 g,
809 ux_scalar,
810 half_n
811 )) goto cleanup;
812
813 /* 2) Store L/R */
814 L_out[r] = Lr;
815 R_out[r] = Rr;
816
817 /* 3) Fiatâ€“Shamir round challenge u_r */
818 if (!derive_ipa_round_challenge(ctx, u_scalar, last_challenge, &Lr, &Rr))
819 goto cleanup;
820
821 /* 4) u_r^{-1} */
822 secp256k1_mpt_scalar_inverse(u_inv, u_scalar);
823 if (!secp256k1_ec_seckey_verify(ctx, u_inv)) goto cleanup;
824
825 /* 5) Update transcript chaining state */
826 memcpy(last_challenge, u_scalar, 32);
827
828 /* 6) Fold vectors in-place */
829 if (!secp256k1_bulletproof_ipa_compress_step(
830 ctx, a_vec, b_vec, G_vec, H_vec, half_n, u_scalar, u_inv
831 )) goto cleanup;
832
833 cur_n = half_n;
834 }
835
836 /* Final folded scalars */
837 memcpy(a_final, a_vec, 32);
838 memcpy(b_final, b_vec, 32);
839
840 if (rounds_out) *rounds_out = rounds;
841 ok = 1;
842
843 cleanup:
844 OPENSSL_cleanse(u_scalar, 32);
845 OPENSSL_cleanse(u_inv, 32);
846 return ok;
847}

849/*
850 * Verifies a Bulletproof Inner Product Argument (IPA).
851 *
852 * Given:
853 * - the original generator vectors G_vec and H_vec,
854 * - the proverâ€™s cross-term commitments L_i and R_i,
855 * - the final folded scalars a_final and b_final,
856 * - the binding scalar ux,
857 * - and the initial commitment P,
858 *
859 * this function re-derives all Fiatâ€“Shamir challenges u_i from the transcript
860 * and reconstructs the folded generators G_f and H_f implicitly.
861 *
862 * Verification checks that the folded commitment P' equals:
863 *
864 * P' = a_final * G_f
865 * + b_final * H_f
866 * + (a_final * b_final * ux) * U
867 *
868 * where G_f and H_f are obtained by folding G_vec and H_vec using the challenges
869 * u_i and their inverses, and P' is obtained by applying the same folding
870 * operations to P using the L_i and R_i commitments.
871 *
872 * All group operations avoid explicit construction of the point at infinity,
873 * which is not representable via the libsecp256k1 public-key API.
874 */
875static int ipa_verify_explicit(
876 const secp256k1_context* ctx,
877 const secp256k1_pubkey* G_vec, /* original G generators (length n) */
878 const secp256k1_pubkey* H_vec, /* original H generators (length n) */
879 const secp256k1_pubkey* U,
880 const secp256k1_pubkey* P_in, /* initial P */
881 const secp256k1_pubkey* L_vec, /* length = rounds */
882 const secp256k1_pubkey* R_vec, /* length = rounds */
883 size_t n, /* total vector length (64*m) */
884 const unsigned char a_final[32],
885 const unsigned char b_final[32],
886 const unsigned char ux[32],
887 const unsigned char ipa_transcript_id[32]
888) {
889 secp256k1_pubkey P = *P_in;
890 secp256k1_pubkey Gf, Hf, RHS, tmp;
891 int RHS_inited = 0;
892 int ok = 0;
893
894 /* --- derive rounds --- */
895 if (n == 0 || (n & (n - 1)) != 0)
896 return 0;
897
898 size_t rounds = bp_ipa_rounds(n);
899
900 /* --- allocate u / u_inv --- */
901 unsigned char* u_flat = (unsigned char*)malloc(rounds * 32);
902 unsigned char* uinv_flat = (unsigned char*)malloc(rounds * 32);
903 if (!u_flat || !uinv_flat)
904 goto cleanup_alloc;
905
906 unsigned char last[32];
907 memcpy(last, ipa_transcript_id, 32);
908
909 /* ---- 1. Re-derive u_i ---- */
910 for (size_t i = 0; i < rounds; i++) {
911 unsigned char* ui = u_flat + 32 * i;
912 unsigned char* uiinv = uinv_flat + 32 * i;
913
914 if (!derive_ipa_round_challenge(ctx, ui, last, &L_vec[i], &R_vec[i]))
915 goto cleanup;
916
917 secp256k1_mpt_scalar_inverse(uiinv, ui);
918 if (!secp256k1_ec_seckey_verify(ctx, uiinv))
919 goto cleanup;
920
921 memcpy(last, ui, 32);
922 }
923
924 /* ---- 2. Fold generators ---- */
925 if (!fold_generators(ctx, &Gf, G_vec, u_flat, uinv_flat, n, rounds, 0))
926 goto cleanup;
927
928 if (!fold_generators(ctx, &Hf, H_vec, u_flat, uinv_flat, n, rounds, 1))
929 goto cleanup;
930
931 /* ---- 3. RHS = a*Gf + b*Hf + (a*b*ux)*U ---- */
932
933 if (!scalar_is_zero(a_final)) {
934 tmp = Gf;
935 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tmp, a_final))
936 goto cleanup;
937 if (!add_term(ctx, &RHS, &RHS_inited, &tmp))
938 goto cleanup;
939 }
940
941 if (!scalar_is_zero(b_final)) {
942 tmp = Hf;
943 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tmp, b_final))
944 goto cleanup;
945 if (!add_term(ctx, &RHS, &RHS_inited, &tmp))
946 goto cleanup;
947 }
948
949 {
950 unsigned char ab[32], ab_ux[32];
951 secp256k1_mpt_scalar_mul(ab, a_final, b_final);
952 secp256k1_mpt_scalar_mul(ab_ux, ab, ux);
953
954 if (!scalar_is_zero(ab_ux)) {
955 tmp = *U;
956 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tmp, ab_ux))
957 goto cleanup;
958 if (!add_term(ctx, &RHS, &RHS_inited, &tmp))
959 goto cleanup;
960 }
961
962 OPENSSL_cleanse(ab, 32);
963 OPENSSL_cleanse(ab_ux, 32);
964 }
965
966 if (!RHS_inited)
967 goto cleanup;
968
969 /* ---- 4. Fold P using L/R ---- */
970 if (!apply_ipa_folding_to_P(
971 ctx, &P, L_vec, R_vec, u_flat, uinv_flat, rounds))
972 goto cleanup;
973
974 /* ---- 5. Compare P and RHS ---- */
975 if (pubkey_equal(ctx, &P, &RHS))
976 ok = 1;
977
978 cleanup:
979 OPENSSL_cleanse(u_flat, rounds * 32);
980 OPENSSL_cleanse(uinv_flat, rounds * 32);
981
982 cleanup_alloc:
983 free(u_flat);
984 free(uinv_flat);
985 return ok;
986}

999int secp256k1_bulletproof_compute_vectors_block(
1000 const secp256k1_context* ctx,
1001 uint64_t value,
1002 size_t block_index, /* j-th value */

1004 unsigned char* ar,
1005 unsigned char* sl,
1006 unsigned char* sr
1007) {
1008 const size_t offset = BP_VALUE_BITS * block_index;
1009 int ok = 1;
1010
1011 /* Scalars */
1012 unsigned char one[32] = {0};
1013 unsigned char minus_one[32];
1014 unsigned char zero[32] = {0};
1015
1016 one[31] = 1;
1017 memcpy(minus_one, one, 32);
1018 secp256k1_mpt_scalar_negate(minus_one, minus_one);
1019

1021 for (size_t i = 0; i < BP_VALUE_BITS; i++) {

1027 memcpy(ar + idx * 32, zero, 32);
1028 } else {
1029 /* bit = 0 => al = 0, ar = -1 */
1030 memcpy(al + idx * 32, zero, 32);
1031 memcpy(ar + idx * 32, minus_one, 32);
1032 }

1036 for (size_t i = 0; i < BP_VALUE_BITS; i++) {

1046 }

1053 OPENSSL_cleanse(al + offset * 32, BP_VALUE_BITS * 32);
1054 OPENSSL_cleanse(ar + offset * 32, BP_VALUE_BITS * 32);
1055 OPENSSL_cleanse(sl + offset * 32, BP_VALUE_BITS * 32);
1056 OPENSSL_cleanse(sr + offset * 32, BP_VALUE_BITS * 32);

1058}

1062int secp256k1_bulletproof_create_commitment(
1063 const secp256k1_context* ctx,
1064 secp256k1_pubkey* commitment_C,
1065 uint64_t value,
1066 const unsigned char* blinding_factor,
1067 const secp256k1_pubkey* pk_base
1068) {
1069 secp256k1_pubkey G_term, Pk_term;
1070 const secp256k1_pubkey* points_to_add[2];
1071 int v_is_zero = (value == 0);
1072
1073 /* 1. Compute r * Pk_base (The Blinding Term) */
1074 Pk_term = *pk_base;
1075 if (secp256k1_ec_pubkey_tweak_mul(ctx, &Pk_term, blinding_factor) != 1) return 0;
1076
1077 /* 2. Handle Value Term */
1078 if (v_is_zero) {
1079 /* If v=0, C = 0*G + r*H = r*H.
1080 We skip G_term entirely because libsecp cannot represent infinity. */
1081 *commitment_C = Pk_term;
1082 return 1;
1083 }
1084
1085 /* 3. Compute v * G (The Value Term) */
1086 if (!compute_amount_point(ctx, &G_term, value)) return 0;
1087
1088 /* 4. Combine: C = v*G + r*Pk_base */
1089 points_to_add[0] = &G_term;
1090 points_to_add[1] = &Pk_term;
1091 if (secp256k1_ec_pubkey_combine(ctx, commitment_C, points_to_add, 2) != 1) return 0;
1092
1093 return 1;
1094}

1218 * A = alpha*Base + <al,G> + <ar,H>
1219 * S = rho*Base + <sl,G> + <sr,H>

1379 if (!generate_random_scalar(ctx, tau2)) goto cleanup;
1380
1381/* ---- 8. Commit T1, T2 ---- */
1382/* T1 = t1*G + tau1*Base where G = G_vec[0] */
1383 {
1384 secp256k1_pubkey tG, tB;
1385 const secp256k1_pubkey* pts[2];
1386
1387 if (memcmp(t1, zero, 32) == 0) goto cleanup; /* prototype guard */
1388 /* tG = t1 * (curve base generator) */
1389 if (!secp256k1_ec_pubkey_create(ctx, &tG, t1)) goto cleanup;
1390
1391
1392 tB = *pk_base;
1393 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tB, tau1)) goto cleanup;
1394
1395 pts[0] = &tG; pts[1] = &tB;
1396 if (!secp256k1_ec_pubkey_combine(ctx, &T1, pts, 2)) goto cleanup;
1397 }
1398
1399/* T2 = t2*G + tau2*Base */
1400 {
1401 secp256k1_pubkey tG, tB;
1402 const secp256k1_pubkey* pts[2];
1403
1404 if (memcmp(t2, zero, 32) == 0) goto cleanup; /* prototype guard */
1405 /* tG = t2 * (curve base generator) */
1406 if (!secp256k1_ec_pubkey_create(ctx, &tG, t2)) goto cleanup;
1407
1408
1409 tB = *pk_base;
1410 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &tB, tau2)) goto cleanup;
1411
1412 pts[0] = &tG; pts[1] = &tB;
1413 if (!secp256k1_ec_pubkey_combine(ctx, &T2, pts, 2)) goto cleanup;
1414 }
1415
1416/* ---- 9. Challenge x ---- */
1417/* x = H(context_id || A || S || y || z || T1 || T2) */
1418 {
1419 unsigned char A_ser[33], S_ser[33], T1_ser[33], T2_ser[33];
1420 size_t len = 33;
1421 SHA256_CTX sha;
1422
1423 if (!secp256k1_ec_pubkey_serialize(ctx, A_ser, &len, &A, SECP256K1_EC_COMPRESSED)) goto cleanup;
1424 len = 33;
1425 if (!secp256k1_ec_pubkey_serialize(ctx, S_ser, &len, &S, SECP256K1_EC_COMPRESSED)) goto cleanup;
1426 len = 33;
1427 if (!secp256k1_ec_pubkey_serialize(ctx, T1_ser, &len, &T1, SECP256K1_EC_COMPRESSED)) goto cleanup;
1428 len = 33;
1429 if (!secp256k1_ec_pubkey_serialize(ctx, T2_ser, &len, &T2, SECP256K1_EC_COMPRESSED)) goto cleanup;
1430
1431 SHA256_Init(&sha);
1432 if (context_id) SHA256_Update(&sha, context_id, 32);
1433 SHA256_Update(&sha, A_ser, 33);
1434 SHA256_Update(&sha, S_ser, 33);
1435 SHA256_Update(&sha, y, 32);
1436 SHA256_Update(&sha, z, 32);
1437 SHA256_Update(&sha, T1_ser, 33);
1438 SHA256_Update(&sha, T2_ser, 33);
1439 SHA256_Final(x, &sha);
1440
1441 secp256k1_mpt_scalar_reduce32(x, x);
1442 if (memcmp(x, zero, 32) == 0) goto cleanup; /* avoid infinity later */
1443 }
1444
1445 /* ---- 10. Evaluate l(x), r(x), t_hat ---- */
1446 for (size_t k = 0; k < n; k++) {
1447 unsigned char tmp[32];
1448
1449 /* l = l0 + sL*x */
1450 secp256k1_mpt_scalar_mul(tmp, sl + 32*k, x);
1451 secp256k1_mpt_scalar_add(l_vec + 32*k, l_vec + 32*k, tmp);
1452
1453 /* r = r0 + r1*x */
1454 secp256k1_mpt_scalar_mul(tmp, r1_vec + 32*k, x);
1455 secp256k1_mpt_scalar_add(r_vec + 32*k, r_vec + 32*k, tmp);
1456
1457 OPENSSL_cleanse(tmp, 32);
1458 }
1459
1460 if (!secp256k1_bulletproof_ipa_dot(ctx, t_hat, l_vec, r_vec, n)) goto cleanup;
1461
1462 /* ---- 11. tau_x and mu (aggregation changes tau_x) ---- */
1463 secp256k1_mpt_scalar_mul(x_sq, x, x);
1464
1465 /* tau_x = tau2*x^2 + tau1*x + sum_j z^(j+2) * blinding_j */
1466 secp256k1_mpt_scalar_mul(tau_x, tau2, x_sq);
1467 {
1468 unsigned char tmp[32];
1469 secp256k1_mpt_scalar_mul(tmp, tau1, x);
1470 secp256k1_mpt_scalar_add(tau_x, tau_x, tmp);
1471
1472 /* + sum_j z^(j+2) * r_j */
1473 for (size_t j = 0; j < m; j++) {
1474 unsigned char add[32];
1475 secp256k1_mpt_scalar_mul(add, z_j2 + 32*j, blindings_flat + 32*j);
1476 secp256k1_mpt_scalar_add(tau_x, tau_x, add);
1477 OPENSSL_cleanse(add, 32);
1478 }
1479
1480 OPENSSL_cleanse(tmp, 32);
1481 }
1482
1483 /* mu = alpha + rho*x */
1484 {
1485 unsigned char tmp[32];
1486 secp256k1_mpt_scalar_mul(tmp, rho, x);
1487 secp256k1_mpt_scalar_add(mu, alpha, tmp);
1488 OPENSSL_cleanse(tmp, 32);
1489 }
1490
1491 /* ---- 12. IPA transcript + ux (binding), and H' normalization ---- */
1492
1493/* 12a. Build a stable IPA transcript seed (32 bytes).
1494 *
1495 * IMPORTANT:
1496 * - Prover and verifier MUST hash the exact same bytes in the exact same order.
1497 * - Use only public elements that both sides already know.
1498 * - Do NOT depend on internal intermediate buffers.
1499 *
1500 * Minimal, safe choice: context_id || A||S||T1||T2 || y||z||x || t_hat
1501 * (All points are serialized compressed 33 bytes.)
1502 */
1503 {
1504 unsigned char A_ser[33], S_ser[33], T1_ser[33], T2_ser[33];
1505 size_t ser_len;
1506 SHA256_CTX sha;
1507
1508 ser_len = 33;
1509 if (!secp256k1_ec_pubkey_serialize(ctx, A_ser, &ser_len, &A, SECP256K1_EC_COMPRESSED)) goto cleanup;
1510 ser_len = 33;
1511 if (!secp256k1_ec_pubkey_serialize(ctx, S_ser, &ser_len, &S, SECP256K1_EC_COMPRESSED)) goto cleanup;
1512 ser_len = 33;
1513 if (!secp256k1_ec_pubkey_serialize(ctx, T1_ser, &ser_len, &T1, SECP256K1_EC_COMPRESSED)) goto cleanup;
1514 ser_len = 33;
1515 if (!secp256k1_ec_pubkey_serialize(ctx, T2_ser, &ser_len, &T2, SECP256K1_EC_COMPRESSED)) goto cleanup;
1516
1517 SHA256_Init(&sha);
1518
1519 /* Domain/context binding */
1520 if (context_id) SHA256_Update(&sha, context_id, 32);
1521
1522 /* Outer commitments */
1523 SHA256_Update(&sha, A_ser, 33);
1524 SHA256_Update(&sha, S_ser, 33);
1525 SHA256_Update(&sha, T1_ser, 33);
1526 SHA256_Update(&sha, T2_ser, 33);
1527
1528 /* Outer challenges */
1529 SHA256_Update(&sha, y, 32);
1530 SHA256_Update(&sha, z, 32);
1531 SHA256_Update(&sha, x, 32);
1532
1533 /* Bind to t_hat as well (public scalar) */
1534 SHA256_Update(&sha, t_hat, 32);
1535
1536 SHA256_Final(ipa_transcript, &sha);
1537 }
1538
1539/* 12b. Derive u_x = H(ipa_transcript || t_hat) reduced to scalar. */
1540 if (!derive_ipa_binding_challenge(ctx, ux_scalar, ipa_transcript, t_hat)) goto cleanup;
1541
1542/* 12c. Normalize H: H'[k] = H[k] * y^{-k}.
1543 *
1544 * NOTE:
1545 * - Requires y != 0.
1546 * - If y==0 (mod q), abort (cannot invert).
1547 */
1548 {
1549 unsigned char y_inv[32];
1550 unsigned char y_inv_pow[32]; /* (y^{-1})^k */
1551 if (memcmp(y, zero, 32) == 0) goto cleanup;
1552
1553 secp256k1_mpt_scalar_inverse(y_inv, y);
1554 memcpy(y_inv_pow, one, 32);
1555
1556 for (size_t k = 0; k < n; k++) {
1557 H_prime[k] = H_vec[k];
1558 /* H_prime[k] = H_vec[k] * (y^{-1})^k */
1559 if (!secp256k1_ec_pubkey_tweak_mul(ctx, &H_prime[k], y_inv_pow)) goto cleanup;
1560 secp256k1_mpt_scalar_mul(y_inv_pow, y_inv_pow, y_inv);
1561 }
1562
1563 OPENSSL_cleanse(y_inv, 32);
1564 OPENSSL_cleanse(y_inv_pow, 32);
1565 }
1566
1567/* 12d. Run IPA prover */
1568 {
1569 size_t rounds_used = 0;
1570
1571 if (!secp256k1_bulletproof_run_ipa_prover(
1572 ctx,
1573 &U, /* binding generator point */
1574 G_vec, /* G */
1575 H_prime, /* H' */
1576 l_vec, /* l(x) scalars (flat 32*n) */
1577 r_vec, /* r(x) scalars (flat 32*n) */
1578 n,
1579 ipa_transcript, /* 32-byte seed */
1580 ux_scalar, /* u_x scalar */
1581 L_vec,
1582 R_vec,
1583 rounds, /* max_rounds = log2(n) */
1584 &rounds_used,
1585 a_final,
1586 b_final
1587 )) goto cleanup;
1588
1589 if (rounds_used != rounds) goto cleanup;
1590 }
1591
1592 /* ---- 13. Serialize (uses rounds) ---- */
1593 {
1594 const size_t expected = 292 + 66 * rounds; /* 4*33 + 2*rounds*33 + 5*32 */
1595
1596 /* Standard pattern: query size only */
1597 if (proof_out == NULL) {
1598 if (proof_len) *proof_len = expected;
1599 ok = 1;
1600 goto cleanup;
1601 }
1602
1603 if (proof_len == NULL) goto cleanup;
1604
1605 if (*proof_len < expected) {
1606 *proof_len = expected;
1607 goto cleanup; /* not enough space */
1608 }
1609
1610 unsigned char* ptr = proof_out;
1611 size_t ser_len;
1612
1613#define SER_PT(P) do { \
1614 ser_len = 33; \
1615 if (!secp256k1_ec_pubkey_serialize(ctx, ptr, &ser_len, &(P), SECP256K1_EC_COMPRESSED)) goto cleanup; \
1616 if (ser_len != 33) goto cleanup; \
1617 ptr += 33; \
1618 } while (0)
1619
1620 SER_PT(A); SER_PT(S); SER_PT(T1); SER_PT(T2);
1621
1622 for (size_t r = 0; r < rounds; r++) SER_PT(L_vec[r]);
1623 for (size_t r = 0; r < rounds; r++) SER_PT(R_vec[r]);
1624
1625 memcpy(ptr, a_final, 32); ptr += 32;
1626 memcpy(ptr, b_final, 32); ptr += 32;
1627 memcpy(ptr, t_hat, 32); ptr += 32;
1628 memcpy(ptr, tau_x, 32); ptr += 32;
1629 memcpy(ptr, mu, 32); ptr += 32;
1630
1631#undef SER_PT
1632
1633 /* Final sanity */
1634 if ((size_t)(ptr - proof_out) != expected) goto cleanup;
1635
1636 *proof_len = expected;
1637 }
1638
1639 ok = 1;
1640
1641 cleanup:
1642
1643 /* wipe sensitive scalars; free buffers */
1644 if (al) OPENSSL_cleanse(al, n * 32);
1645 if (ar) OPENSSL_cleanse(ar, n * 32);
1646 if (sl) OPENSSL_cleanse(sl, n * 32);
1647 if (sr) OPENSSL_cleanse(sr, n * 32);
1648 if (l_vec) OPENSSL_cleanse(l_vec, n * 32);
1649 if (r_vec) OPENSSL_cleanse(r_vec, n * 32);
1650 if (r1_vec) OPENSSL_cleanse(r1_vec, n * 32);
1651
1652 OPENSSL_cleanse(alpha, 32);
1653 OPENSSL_cleanse(rho, 32);
1654 OPENSSL_cleanse(tau1, 32);
1655 OPENSSL_cleanse(tau2, 32);
1656 OPENSSL_cleanse(t1, 32);
1657 OPENSSL_cleanse(t2, 32);
1658 OPENSSL_cleanse(t_hat, 32);
1659 OPENSSL_cleanse(tau_x, 32);
1660 OPENSSL_cleanse(mu, 32);
1661
1662 OPENSSL_cleanse(y, 32);
1663 OPENSSL_cleanse(z, 32);
1664 OPENSSL_cleanse(x, 32);
1665 OPENSSL_cleanse(z_sq, 32);
1666 OPENSSL_cleanse(z_neg, 32);
1667 OPENSSL_cleanse(x_sq, 32);
1668 OPENSSL_cleanse(ux_scalar, 32);
1669 OPENSSL_cleanse(ipa_transcript, 32);
1670
1671 if (G_vec) { OPENSSL_cleanse(G_vec, n * sizeof(secp256k1_pubkey)); free(G_vec); }
1672 if (H_vec) { OPENSSL_cleanse(H_vec, n * sizeof(secp256k1_pubkey)); free(H_vec); }
1673 if (H_prime) { OPENSSL_cleanse(H_prime, n * sizeof(secp256k1_pubkey)); free(H_prime); }
1674
1675 if (al) free(al);
1676 if (ar) free(ar);
1677 if (sl) free(sl);
1678 if (sr) free(sr);
1679 if (l_vec) free(l_vec);
1680 if (r_vec) free(r_vec);
1681 if (r1_vec) free(r1_vec);
1682
1683 if (L_vec) { OPENSSL_cleanse(L_vec, rounds * sizeof(secp256k1_pubkey)); free(L_vec); }
1684 if (R_vec) { OPENSSL_cleanse(R_vec, rounds * sizeof(secp256k1_pubkey)); free(R_vec); }
1685
1686 if (y_powers) { OPENSSL_cleanse(y_powers, n * 32); free(y_powers); }
1687 if (z_j2) { OPENSSL_cleanse(z_j2, m * 32); free(z_j2); }
1688
1689 return ok;
1690}

1716int secp256k1_bulletproof_verify_agg(
1717 const secp256k1_context* ctx,
1718 const secp256k1_pubkey* G_vec, /* length n = 64*m */
1719 const secp256k1_pubkey* H_vec, /* length n = 64*m */
1720 const unsigned char* proof,
1721 size_t proof_len,
1722 const secp256k1_pubkey* commitment_C_vec, /* length m */
1723 size_t m,
1724 const secp256k1_pubkey* pk_base,
1725 const unsigned char* context_id
1726) {
1727
1728 if (!ctx || !G_vec || !H_vec || !proof || !commitment_C_vec || !pk_base) return 0;
1729 if (m == 0) return 0;
1730 /* Aggregation requires n = 64*m to be power-of-two => m must be power-of-two. */
1731 if ((m & (m - 1)) != 0) return 0;
1732
1733 const size_t n = BP_TOTAL_BITS(m); /* 64*m */
1734 const size_t rounds = bp_ipa_rounds(n);
1735
1736 /* Proof length is dynamic in aggregated mode */
1737 const size_t expected_len = 292 + 66 * rounds;
1738 if (proof_len != expected_len) return 0;
1739
1740 /* --- Unpack proof --- */
1741 secp256k1_pubkey A, S, T1, T2;
1742 secp256k1_pubkey U;
1743
1744 secp256k1_pubkey* L_vec = (secp256k1_pubkey*)malloc(rounds * sizeof(secp256k1_pubkey));
1745 secp256k1_pubkey* R_vec = (secp256k1_pubkey*)malloc(rounds * sizeof(secp256k1_pubkey));
1746 if (!L_vec || !R_vec) { free(L_vec); free(R_vec); return 0; }
1747
1748 unsigned char a_final[32], b_final[32];
1749 unsigned char t_hat[32], tau_x[32], mu[32];
1750
1751 unsigned char y[32], z[32], x[32];
1752 unsigned char ux_scalar[32];
1753 unsigned char z_sq[32];
1754
1755 const unsigned char* ptr = proof;
1756
1757 if (!secp256k1_ec_pubkey_parse(ctx, &A, ptr, 33)) goto fail; ptr += 33;
1758 if (!secp256k1_ec_pubkey_parse(ctx, &S, ptr, 33)) goto fail; ptr += 33;
1759 if (!secp256k1_ec_pubkey_parse(ctx, &T1, ptr, 33)) goto fail; ptr += 33;
1760 if (!secp256k1_ec_pubkey_parse(ctx, &T2, ptr, 33)) goto fail; ptr += 33;
1761
1762 for (size_t i = 0; i < rounds; i++) {
1763 if (!secp256k1_ec_pubkey_parse(ctx, &L_vec[i], ptr, 33)) goto fail;
1764 ptr += 33;
1765 }
1766
1767 for (size_t i = 0; i < rounds; i++) {
1768 if (!secp256k1_ec_pubkey_parse(ctx, &R_vec[i], ptr, 33)) goto fail;
1769 ptr += 33;
1770 }
1771
1772 memcpy(a_final, ptr, 32); ptr += 32;
1773 memcpy(b_final, ptr, 32); ptr += 32;
1774 memcpy(t_hat, ptr, 32); ptr += 32;
1775 memcpy(tau_x, ptr, 32); ptr += 32;
1776 memcpy(mu, ptr, 32); ptr += 32;
1777
1778 /* Basic scalar validity */
1779 if (!secp256k1_ec_seckey_verify(ctx, a_final)) goto fail;
1780 if (!secp256k1_ec_seckey_verify(ctx, b_final)) goto fail;
1781 if (!secp256k1_ec_seckey_verify(ctx, t_hat)) goto fail;
1782 if (!secp256k1_ec_seckey_verify(ctx, tau_x)) goto fail;
1783 if (!secp256k1_ec_seckey_verify(ctx, mu)) goto fail;
1784
1785 /* Derive U */
1786 {
1787 secp256k1_pubkey U_arr[1];
1788 if (!secp256k1_mpt_get_generator_vector(ctx, U_arr, 1, (const unsigned char*)"BP_U", 4)) goto fail;
1789 U = U_arr[0];
1790 }
1791
1792 /* --- Fiatâ€“Shamir: y,z --- */
1793 unsigned char A_ser[33], S_ser[33], T1_ser[33], T2_ser[33];
1794 size_t slen = 33;
1795 SHA256_CTX sha;
1796
1797 if (!secp256k1_ec_pubkey_serialize(ctx, A_ser, &slen, &A, SECP256K1_EC_COMPRESSED)) goto fail;
1798 slen = 33;
1799 if (!secp256k1_ec_pubkey_serialize(ctx, S_ser, &slen, &S, SECP256K1_EC_COMPRESSED)) goto fail;
1800 slen = 33;
1801 if (!secp256k1_ec_pubkey_serialize(ctx, T1_ser, &slen, &T1, SECP256K1_EC_COMPRESSED)) goto fail;
1802 slen = 33;
1803 if (!secp256k1_ec_pubkey_serialize(ctx, T2_ser, &slen, &T2, SECP256K1_EC_COMPRESSED)) goto fail;
1804
1805 SHA256_Init(&sha);
1806
1807 /* 1. Domain Separation */
1808 SHA256_Update(&sha, "MPT_BULLETPROOF_RANGE", 21);
1809
1810 /* 2. Transaction Context */
1811 if (context_id) SHA256_Update(&sha, context_id, 32);
1812
1813 /* 3. Value Commitments (Inputs to Verify) */
1814 for (size_t i = 0; i < m; i++) {
1815 unsigned char C_ser[33];
1816 size_t c_len = 33;
1817 if (!secp256k1_ec_pubkey_serialize(ctx, C_ser, &c_len, &commitment_C_vec[i], SECP256K1_EC_COMPRESSED)) goto fail;
1818 SHA256_Update(&sha, C_ser, 33);
1819 }
1820
1821 /* 4. A, S */
1822 SHA256_Update(&sha, A_ser, 33);
1823 SHA256_Update(&sha, S_ser, 33);
1824 SHA256_Final(y, &sha);
1825 secp256k1_mpt_scalar_reduce32(y, y);
1826
1827 /* Generate z (Hash T1 || y) */
1828 SHA256_Init(&sha);
1829 SHA256_Update(&sha, "MPT_BULLETPROOF_RANGE", 21);
1830 if (context_id) SHA256_Update(&sha, context_id, 32);
1831 for (size_t i = 0; i < m; i++) {
1832 unsigned char C_ser[33];
1833 size_t c_len = 33;
1834 secp256k1_ec_pubkey_serialize(ctx, C_ser, &c_len, &commitment_C_vec[i], SECP256K1_EC_COMPRESSED);
1835 SHA256_Update(&sha, C_ser, 33);
1836 }
1837 SHA256_Update(&sha, A_ser, 33);
1838 SHA256_Update(&sha, S_ser, 33);
1839 SHA256_Update(&sha, y, 32);
1840 SHA256_Final(z, &sha);
1841 secp256k1_mpt_scalar_reduce32(z, z);
1842
1843 if (!secp256k1_ec_seckey_verify(ctx, y) || !secp256k1_ec_seckey_verify(ctx, z)) goto fail;
1844
1845 /* Powers */
1846 unsigned char* y_powers = (unsigned char*)malloc(n * 32);
1847 unsigned char* y_inv_powers = (unsigned char*)malloc(n * 32);
1848 if (!y_powers || !y_inv_powers) { free(y_powers); free(y_inv_powers); goto fail; }
1849
1850 unsigned char y_inv[32];
1851 scalar_vector_powers(ctx, (unsigned char (*)[32])y_powers, y, n);
1852 secp256k1_mpt_scalar_inverse(y_inv, y);
1853 scalar_vector_powers(ctx, (unsigned char (*)[32])y_inv_powers, y_inv, n);
1854
1855 /* --- Fiatâ€“Shamir: x --- */
1856 SHA256_Init(&sha);
1857 if (context_id) SHA256_Update(&sha, context_id, 32);
1858 SHA256_Update(&sha, A_ser, 33);
1859 SHA256_Update(&sha, S_ser, 33);
1860 SHA256_Update(&sha, y, 32);
1861 SHA256_Update(&sha, z, 32);
1862 SHA256_Update(&sha, T1_ser, 33);
1863 SHA256_Update(&sha, T2_ser, 33);
1864 SHA256_Final(x, &sha);
1865 secp256k1_mpt_scalar_reduce32(x, x);
1866
1867 if (!secp256k1_ec_seckey_verify(ctx, x)) { free(y_powers); free(y_inv_powers); goto fail; }
1868
1869 /* z^2 */
1870 secp256k1_mpt_scalar_mul(z_sq, z, z);
1871
1872 /* =========================================================================
1873 * Step 3: Verify polynomial identity:
1874 * t_hat*G + tau_x*H == (sum_j z^(j+2) * V_j) + delta(y,z)*G + x*T1 + x^2*T2
1875 * ========================================================================= */
1876
1877 /* --- delta(y,z) for aggregation --- */
1878 unsigned char (*y_block_sum)[32] = (unsigned char (*)[32])malloc(m * 32);
1879 if (!y_block_sum) { free(y_powers); free(y_inv_powers); goto fail; }
1880
1881 unsigned char two_sum[32];
1882 compute_delta_scalars(ctx, y_block_sum, two_sum, y, m);
1883
1884 unsigned char delta[32] = {0};
1885 unsigned char sum_y_all[32] = {0};
1886
1887/* sum_y_all = sum_{k=0}^{n-1} y^k = sum_j y_block_sum[j] */
1888 for (size_t j = 0; j < m; j++) {
1889 secp256k1_mpt_scalar_add(sum_y_all, sum_y_all, y_block_sum[j]);
1890 }
1891
1892/* delta += (z - z^2) * sum_y_all */
1893 {
1894 unsigned char z_minus_z2[32], tmp[32];
1895 secp256k1_mpt_scalar_sub(z_minus_z2, z, z_sq);
1896 secp256k1_mpt_scalar_mul(tmp, z_minus_z2, sum_y_all);
1897 secp256k1_mpt_scalar_add(delta, delta, tmp);
1898 OPENSSL_cleanse(z_minus_z2, 32);
1899 OPENSSL_cleanse(tmp, 32);
1900 }
1901
1902/* delta -= sum_{j=0}^{m-1} z^(j+3) * two_sum */
1903 for (size_t j = 0; j < m; j++) {
1904 unsigned char z_j3[32], tmp[32];
1905 scalar_pow_u32(ctx, z_j3, z, (unsigned int)(j + 3));
1906 secp256k1_mpt_scalar_mul(tmp, z_j3, two_sum);
1907 secp256k1_mpt_scalar_negate(tmp, tmp);
1908 secp256k1_mpt_scalar_add(delta, delta, tmp);
1909 OPENSSL_cleanse(z_j3, 32);
1910 OPENSSL_cleanse(tmp, 32);
1911 }
1912
1913 OPENSSL_cleanse(sum_y_all, 32);
1914
1915
